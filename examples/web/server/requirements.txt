# The origo Python bindings with the FastAPI extra, plus an ASGI server.
#
# From a checkout you build the bindings locally instead of installing from an
# index — the extension is a compiled pyo3 module:
#
#     (cd ../../../crates/origo-py && python -m venv .venv && . .venv/bin/activate \
#        && pip install maturin && maturin develop)
#     pip install fastapi "uvicorn[standard]"
#
# When origo is published, this file is all you need:
origo[fastapi]
uvicorn[standard]
